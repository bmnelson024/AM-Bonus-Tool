        for(let c=row.length-1;c>=0;c--){const v=parseFloat(String(row[c]).replace(/,/g,''));if(!isNaN(v))return v;}

        return null;

      }


      for(let i=headerRowIdx+1;i<data.length;i++){

        const row=data[i];

        const label=String(row[0]||'').trim().toLowerCase();

        for(const[key,patterns] of Object.entries(LABELS)){

          if(items[key]===null && patterns.some(p=>label===p||label===p.trim()||label.includes(p))){

function parseStatement(file, type, rid){

  const reader=new FileReader();

  reader.onload=e=>{

    try {

      const wb=XLSX.read(e.target.result,{type:'array'});

      const ws=wb.Sheets[wb.SheetNames[0]];

      const data=XLSX.utils.sheet_to_json(ws,{header:1,defval:''});


      const periodMonths = getPeriodMonths(type);


      let platform='', propName='', headerRow=null, headerRowIdx=-1;

      let noi=null, cf=null;


      // ── Detect platform and find headers ──

      for(let i=0;i<Math.min(data.length,15);i++){

        const row=data[i];

        const r0=String(row[0]||'').trim();

        const r1=String(row[1]||'').trim();


        // Appfolio: "Properties:" in col A or B

        if(r0.startsWith('Properties:')||r1.startsWith('Properties:')){

          platform='Appfolio';

          const raw=(r0.startsWith('Properties:')?r0:r1).replace('Properties:','').trim();

          const parts=raw.split(' - ');

          propName=parts.length>=2?parts[1].trim():raw;

        }

        // Buildium: property name is a standalone row (row 2, col A)

        if(i===2 && !platform && r0 && !r0.includes('Income Statement') && !r0.includes('Prepared')){

          platform='Buildium';

          // Strip extra identifiers: "Bancroft Place Apartments - SD - PGO - ..."

          const parts=r0.split(' - ');

          propName=parts[0].trim();

        }


        // Find the column header row (contains month-like values)

        if(!headerRow && row.some(c=>String(c).match(/^\d{2}-\d{4}$|^(Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec)\s\d{4}$/))){

          headerRow=row;

          headerRowIdx=i;

        }

      }


      if(!platform) platform='Unknown';


      // ── Find which columns correspond to our period ──

      let targetCols=[];

      if(headerRow && periodMonths){

        headerRow.forEach((h,ci)=>{

          if(periodMonths.some(pm=>monthMatchesHeader(h,pm.year,pm.month))){

            targetCols.push(ci);

          }

        });

      }


      const periodLabel = periodMonths

        ? `${MONTH_NAMES[periodMonths[0].month-1]} ${periodMonths[0].year} – ${MONTH_NAMES[periodMonths[periodMonths.length-1].month-1]} ${periodMonths[periodMonths.length-1].year} (${targetCols.length} months)`

        : 'Full period (no date filter)';


      // ── Scan rows for NOI and Net Income ──

      for(let i=headerRowIdx+1;i<data.length;i++){

        const row=data[i];

        const label=String(row[0]||'').trim().toLowerCase();


        // Match NOI row

        const isNOI = label==='net operating income' ||

                      label==='noi - net operating income' ||

                      (label.includes('noi') && label.includes('net operating income'));


        // Match Net Income row — must be exact, avoid "Net Non-operating Income" etc.

        const isNI = label==='net income';


        if((isNOI && noi===null)||(isNI && cf===null)){

          let val=null;

          if(targetCols.length>0){

            // Sum only the target period columns

            val=targetCols.reduce((s,ci)=>{

              const v=parseFloat(String(row[ci]).replace(/,/g,''));

              return s+(isNaN(v)?0:v);

            },0);

          } else {

            // Fallback: use last numeric value (Total column)

            for(let c=row.length-1;c>=0;c--){

              const v=parseFloat(String(row[c]).replace(/,/g,''));

              if(!isNaN(v)){val=v;break;}

            }

          }

          if(isNOI && noi===null) noi=val;

          if(isNI  && cf===null)  cf=val;

        }

        if(noi!==null && cf!==null) break;

      }


      pendingImport={type,rowId:rid,noi,cf,propName,platform,periodLabel};

      showModal();

    } catch(err){

      alert('Could not read the file. Please make sure it is a valid Appfolio or Buildium Excel export.\n\n'+err.message);

    }

  };

  reader.readAsArrayBuffer(file);

}

        ? `<button class="upload-btn-cf ${row.cfImported?'loaded':''}" onclick="triggerCFUpload('${type}',${row.id})">${cfUploadIcon}${row.cfImported?'CF Imported':'Import CF stmt'}</button>`

        : '<span style="font-size:11px;color:var(--text3)">—</span>'

      }</td>

      <td class="${b>0?'bonus-val':'zero-val'}">${fmtD(b,2)}</td>

      <td><button class="del-btn" onclick="removeRow('${type}',${row.id})">×</button></td>

    </tr>`;

  });

  const capped=totalBonus>pool, displayTotal=Math.min(totalBonus,pool);

}

function renderGroups(type){

  const cont=document.getElementById('groups-'+type);

  if(!state[type].groups.length){cont.innerHTML=`<p class="empty-state">No syndication groups.</p>`;return;}

  cont.innerHTML=state[type].groups.map(g=>{

    const props=state[type].rows.filter(r=>r.group===g.name);

    const totalCF=props.reduce((s,r)=>s+(parseFloat(r.adjCf!==''&&r.adjCf!==undefined?r.adjCf:r.cf)||0),0);

    const k=parseFloat(g.capital);

    const pct=(!isNaN(k)&&k!==0)?(totalCF*2/k*100):null;

    const propNames=props.map(p=>p.name||'(unnamed)').join(', ')||'No properties assigned';

    return`<div class="group-row"><div class="group-head">

    </div><div class="group-props">Properties: ${propNames}</div></div>`;

}


// ─── results ──────────────────────────────────────────────────────────────────

function calcEarned(type){

  const pool=parseFloat(document.getElementById('pool-'+type).value)||0;

  const n=state[type].rows.length;if(!n)return 0;

  const rate=pool/n, cfMap=buildCFMap(type);

  const raw=state[type].rows.reduce((s,r)=>{

    const eCF=r.adjCf!==''&&r.adjCf!==undefined?r.adjCf:r.cf;

    const ot=occTier(r.occ),nt=noiTier(r.priorNoi,r.currentNoi),ct=r.group?(cfMap[r.group]||'miss'):cfTierFromPct(annPct(eCF,r.capital));

    return s+rate*(wOcc(ot)+wNOI(nt)+wCF(ct));

  },0);

  return Math.min(raw,pool);

}
